81      private void gitSiteDeploy() throws MojoExecutionException, MojoFailureException {
82          try {
83              // perform the push
84              final String gitRemoteUrl = gitScmUrl.substring(SCM_PREFIX.length());
85              FileUtils.deleteDirectory(new File(inputDirectory, ".git"));
86              FileUtils.fileWrite(new File(inputDirectory, ".gitattributes").getAbsolutePath(), "* text=auto\n");
87              final int fileCount = FileUtils.getFiles(inputDirectory, null, null, false).size();
88              if (keepHistory) {
89                  // branch must already exist! TODO: detect that
90                  final String origDir = inputDirectory.getAbsolutePath() + ".orig";
91                  git("clone", "--branch", gitBranch, "--single-branch", gitRemoteUrl, origDir);
92                  FileUtils.rename(new File(origDir, ".git"), new File(inputDirectory, ".git"));
93                  git("add", "-A", ".");
94                  git("commit", "-am", String.format("Updating site with %d files", fileCount));
95                  git("push", gitRemoteUrl);
96              } else {
97                  git("init");
98                  git("add", "-A", ".");
99                  git("commit", "-am", String.format("Creating site with %d files", fileCount));
100                 git("push", gitRemoteUrl, "+master:" + gitBranch);
101             }
102         } catch (CommandLineException e) {
103             throw new MojoExecutionException("git publishing error", e);
104         } catch (IOException e) {
105             throw new MojoExecutionException("git publishing error", e);
106         }
107     }

109     private void git(String... args) throws CommandLineException, MojoFailureException {
110         final Commandline cl = new Commandline();
111         cl.setWorkingDirectory(inputDirectory);
112         cl.setExecutable("git");
113         cl.addArguments(args);
114         final StreamConsumer stdout = new StreamConsumer() {
115             public void consumeLine(String line) {
116                 debug(line);
117             }
118         };
119         final StreamConsumer stderr = new StreamConsumer() {
120             public void consumeLine(String line) {
121                 warn(line);
122             }
123         };
124         info(String.format("Executing: %s", cl));
125         final int exitCode = CommandLineUtils.executeCommandLine(cl, stdout, stderr);
126         if (exitCode != 0) {
127             throw new MojoFailureException(String.format("git returned with exit code '%d'", exitCode));
128         }
129     }

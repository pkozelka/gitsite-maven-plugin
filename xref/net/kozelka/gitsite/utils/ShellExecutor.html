3   import java.io.File;
4   import java.util.ArrayList;
5   import java.util.List;
6   import org.codehaus.plexus.util.cli.CommandLineException;
7   import org.codehaus.plexus.util.cli.CommandLineUtils;
8   import org.codehaus.plexus.util.cli.Commandline;
9   import org.codehaus.plexus.util.cli.DefaultConsumer;
10  import org.codehaus.plexus.util.cli.StreamConsumer;
11  
12  public class ShellExecutor {
13      private File workingDirectory;
14      private StreamConsumer info = new DefaultConsumer();
15      private StreamConsumer stdout = new DefaultConsumer();
16      private StreamConsumer stderr = new DefaultConsumer();
17  
18      public void setWorkingDirectory(File workingDirectory) {
19          this.workingDirectory = workingDirectory;
20      }
21  
22      public void setInfo(StreamConsumer info) {
23          this.info = info;
24      }
25  
26      public void setStdout(StreamConsumer stdout) {
27          this.stdout = stdout;
28      }
29  
30      public void setStderr(StreamConsumer stderr) {
31          this.stderr = stderr;
32      }
33  
34      public void exec(String executable, String... args) throws CommandLineException {
35          final Commandline cl = new Commandline();
36          if (workingDirectory != null) {
37              cl.setWorkingDirectory(workingDirectory);
38          }
39          cl.setExecutable(executable);
40          cl.addArguments(args);
41          info.consumeLine(String.format("Executing: %s", cl.toString()));
42          final int exitCode = CommandLineUtils.executeCommandLine(cl, stdout, stderr);
43          if (exitCode != 0) {
44              throw new CommandLineException(String.format("%s returned with exit code '%d'", executable, exitCode));
45          }
46      }
47  
48      public void execWithResult(final Result result, String executable, String... args) throws CommandLineException {
49          final Commandline cl = new Commandline();
50          cl.setWorkingDirectory(workingDirectory);
51          cl.setExecutable(executable);
52          cl.addArguments(args);
53          info.consumeLine(String.format("Executing: %s", cl));
54          result.exitCode = CommandLineUtils.executeCommandLine(cl,
55              new StreamConsumer() {
56                  public void consumeLine(String line) {
57                      stdout.consumeLine(line);
58                      result.stdoutLines.add(line);
59                  }
60              },
61              new StreamConsumer() {
62                  public void consumeLine(String line) {
63                      stderr.consumeLine(line);
64                      result.stderrLines.add(line);
65                  }
66              });
67          info.consumeLine(String.format("%s exited with code %d", executable, result.exitCode));
68      }
69  
70      public static class Result {
71          private int exitCode;
72          private final List<String> stdoutLines = new ArrayList<String>();
73          private final List<String> stderrLines = new ArrayList<String>();
74  
75          public int getExitCode() {
76              return exitCode;
77          }
78  
79          public List<String> getStdoutLines() {
80              return stdoutLines;
81          }
82  
83          public List<String> getStderrLines() {
84              return stderrLines;
85          }
86      }
87  }

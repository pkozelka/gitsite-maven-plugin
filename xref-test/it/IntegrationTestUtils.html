7   import org.codehaus.plexus.util.FileUtils;
8   import org.codehaus.plexus.util.cli.CommandLineException;
9   
10  class IntegrationTestUtils {
11  
12      static void createGitRepo(File gitwc, String branchToPrefill) throws CommandLineException, IOException {
13          final ShellExecutor shell = new ShellExecutor();
14          shell.exec("git", "init", gitwc.getAbsolutePath());
15          final File helloTxtFile = new File(gitwc, "Hello.txt");
16          FileUtils.fileWrite(helloTxtFile, "Hello world\n");
17          shell.setWorkingDirectory(gitwc);
18          shell.exec("git", "add", "-A");
19          shell.exec("git", "commit", "-am", "initial commit");
20          shell.exec("git", "checkout", "-b", branchToPrefill);
21          for (int i=1; i<=5; i++) {
22              FileUtils.fileAppend(helloTxtFile.getAbsolutePath(),
23                  String.format("%s commit %d into %s%n",
24                      new Date(), i, branchToPrefill));
25              shell.exec("git", "commit", "-am", "commit no." + i);
26          }
27          shell.exec("git", "config", "receive.denyCurrentBranch", "warn");
28      }
29  }
